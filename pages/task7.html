Задача 7

Создать функцию, рассчитывающую факториал заданного числа минимум тремя разными способами. Число и тип данных вывести на экран.

Решение

document.getElementById('check').onclick = function() {
	calculateFactorial(document.getElementById('input').value);
};

// Вычисление факториала числа
// @param incomingString (string) - строка, из которой нужно вырезать число, факториал которого мы высчитываем

function calculateFactorial (incomingString) {
	var num = parseInt(incomingString);
	for (var i = incomingString - 1; i > 0; i--) {
		num *= i;
	}
	alert(num);
}

или

document.getElementById('check').onclick = function() {
	alert(calculateFactorial(document.getElementById('input').value));
};

// Вычисление факториала числа
// @param incomingString (string) - строка, из которой нужно вырезать число, факториал которого мы высчитываем

function calculateFactorial (incomingString) {
  if (incomingString != 1) {
  	return incomingString * calculateFactorial(incomingString - 1);
  }

  else {
  	return 1;
  }
}

document.getElementById('check').onclick = function() {

// Вычисление факториала числа
// @param incomingString (string) - строка, из которой нужно вырезать число, факториал которого мы высчитываем

function calculateFactorial (incomingString) {
  return incomingString ? incomingString * calculateFactorial(incomingString - 1) : 1;
}
