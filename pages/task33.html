Задача 33

Дан массив ['Здоровенная', 'порция', 'сарказма', 'и', 'нецензурщины']. С помощью метода join преобразуйте массив в строку «Здоровенная порция сарказма и нецензурщины». Попереставлять слова местами.

Решение

var arr = ['Здоровенная', 'порция', 'сарказма', 'и', 'нецензурщины'],
    string = arr.join(' ');
document.getElementById('check').onclick = function () {
    var newElem = document.createElement('p');
    document.getElementById('result').innerHTML = '';
    newElem.innerHTML = string;
    document.getElementById('result').appendChild(newElem);
};

//Чтобы перемешать слова местами, используем функцию mixRandom из задачи №27

var arr = ['Здоровенная', 'порция', 'сарказма', 'и', 'нецензурщины'],
    string = '';

arr = arr.sort(mixRandom);
string = arr.join(' ');
