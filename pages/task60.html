Задача 60

С помощью функции нарисуй треугольник Паскаля:

1
11
121
1331
14641
151051

Довести количество итераций до 100.

Решение

document.getElementById('check').onclick = function () {
    drawPascalTriangle();
};

function drawPascalTriangle() {
    var arr = [];

    document.write ('<p>');
    for(var i=1; i<=100 ; i++) {
        arr[i] = 0;
    }
    arr[0] = 1;

    for(var j=1;j<=100; j++) {
        for(var k=j; k>=1; k--) {
            document.write (arr[k-1]);
            arr[k] = arr[k-1] + arr[k];
        }
        document.write('<p>');
    }
}
