Задача 14

Создайте объект минимум тремя разными способами. Выведите его на экран.

Решение

var cat = {
    name: 'Пуркович',
    age: 5,
    color: 'Рыжий',
};

или

var cat = new Object();
cat.name = 'Пуркович';
cat.age = 5;
cat.color = 'Рыжий';

или

var cat = new Cat('Пуркович', 5, 'Рыжий');

function Cat(name, age, color) {
    this.name = name;
    this.age = age;
    this.color = color;
}

document.getElementById('check').onclick = function() {
    alert('Новый котик ' + cat.name + '.  Его возраст: ' + cat.age + ' лет. Он ' + cat.color + '.');
}
