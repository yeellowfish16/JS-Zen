Задача 82

Разбиваем число пополам и сравниваем сумму первой его половины с суммой второй половины, например, для строки 16742862 вернуть true, т.к. 1+6+7+4=18 и 2+8+6+2=18. В случае, если в строке нечетное количество цифр, вернуть "В строке нечетное количество цифр".

Решение

document.getElementById('check').onclick = function () {
    var newElem = document.createElement('p');
    document.getElementById('result').innerHTML = '';
    newElem.innerHTML = compareHalfOfNumber(document.getElementById('input').value);
    document.getElementById('result').appendChild(newElem);
};

// Функция для сравнения суммы первых цифр числа и последних

function compareHalfOfNumber(string) {
    var array = string.split(''),
        sum1 = 0,
        sum2 = 0;

    // Используем функцию isEven из задачи 11
    if (!isEven(array.length)) {
        return 'В строке нечетное количество цифр';
    }

    for (var i = 0; i < array.length/2; i++) {
        sum1 += Number(array[i]);
    }

    for (var j = array.length/2; j < array.length; j++) {
        sum2 += Number(array[j]);
    }

    return sum1 == sum2 ? true : false;
}
