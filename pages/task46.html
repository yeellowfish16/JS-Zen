Задача 46

Дана строка. Сделайте заглавным первый символ этой строки не используя цикл. Найдите два решения.

Решение

var string = 'все счастливые семьи похожи друг на друга, каждая несчастливая семья несчастлива по-своему.';

//1 вариант

string = string[0].toUpperCase() + string.slice(1);

//2 вариант

var arr = string.split('');

arr[0] = arr[0].toUpperCase();
string = arr.join('');

document.getElementById('check').onclick = function () {
    var newElem = document.createElement('p');
    document.getElementById('result').innerHTML = '';
    newElem.innerHTML = string;
    document.getElementById('result').appendChild(newElem);
};
